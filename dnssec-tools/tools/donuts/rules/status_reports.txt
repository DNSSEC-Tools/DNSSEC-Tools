  use Date::Parse;
  $donuts::time = time();

  $donuts::sigexptime = undef;
</init>
<test>
  my $date = $record->sigexpiration;
  $date =~ s/(....)(..)(..)(..)(..)(..)/$2-$3-$1 $4:$5:$6/;
  my $extime = str2time($date);

  if (!defined($donuts::sigexptime)) {
    $donuts::sigexptime = $record->sigexpiration;
  } elsif ($donuts::sigexptime != $record->sigexpiration) {
    donuts_error("warning: signatures in this zone are not time aligned");
    
    $donuts::sigexptime = $donuts::sigexptime < $record->sigexpiration
                              ? $donuts::sigexptime : $record->sigexpiration;
  } 

</test>
