
usage:  rollctl [options] 
	-halt				shutdown rollerd
	-display			start graphical display
	-dspub <zone>			parent has published DS record for zone
	-dspuball			parents have published DS records for zones
	-logfile <logfile>		set log file
	-loglevel <loglevel>		set logging level
	-nodisplay			stop graphical display
	-rollall			roll all zones
	-rollksk <rollrec>		roll specified zone's KSK
	-rollrec <rollrec>		set rollrec file
	-rollzone <zone>		roll named zone
	-runqueue			run queue
	-shutdown			shutdown rollerd
	-skipall <zone>			skip named zone
	-skipzone <zone>		skip named zone
	-sleeptime <sleeptime>		set sleep time
	-status				get rollerd's status
	-zonelog			set a zone's log level
	-zonestatus			get status of zones
	-Version			display version number
	-quiet				don't give any output
	-help				help message
